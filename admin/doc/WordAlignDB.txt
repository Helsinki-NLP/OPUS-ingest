CREATE TABLE IF NOT EXISTS `tiedeman`.`eng-dut` (
`eng` VARCHAR( 100 ) CHARACTER SET utf8 COLLATE utf8_unicode_ci NOT NULL ,
`dut` VARCHAR( 100 ) CHARACTER SET utf8 COLLATE utf8_unicode_ci NOT NULL ,
`freq` INT NOT NULL ,
`originID` INT NOT NULL ,
`statusID` INT NOT NULL ,
INDEX ( `eng` , `dut` , `freq` , `originID` , `statusID` )
)ENGINE = MyISAM



 CREATE TABLE IF NOT EXISTS `tiedeman`.`origin` (
`originID` INT NOT NULL AUTO_INCREMENT PRIMARY KEY ,
`corpus` VARCHAR( 50 ) NOT NULL ,
`alignment` VARCHAR( 100 ) NOT NULL ,
`date` DATE NOT NULL ,
UNIQUE INDEX ( `corpus` , `alignment` , `date` )
) ENGINE = MYISAM


 CREATE TABLE IF NOT EXISTS `tiedeman`.`status` (
`statusID` INT NOT NULL AUTO_INCREMENT PRIMARY KEY ,
`status` VARCHAR( 30 ) NOT NULL ,
`userID` INT NOT NULL ,
UNIQUE INDEX ( `status` , `userID` )
) ENGINE = MYISAM



 CREATE TABLE IF NOT EXISTS `tiedeman`.`user` (
`userID` INT NOT NULL AUTO_INCREMENT PRIMARY KEY ,
`name` VARCHAR( 100 ) NOT NULL UNIQUE,
`info` VARCHAR( 200 ) NOT NULL ,
INDEX ( `name` )
) ENGINE = MYISAM
