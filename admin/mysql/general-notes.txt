

* escape table names with hyphens: use backticks, e.g.
  DESCRIBE `spa-swe`

* add a new database and grant permissions to a specific user
  example: database 'lexicon' and grant permissions for user 'opus'

CREATE DATABASE lexicon;
show grants for 'opus'@'localhost';
GRANT SELECT, INSERT, UPDATE, DELETE, CREATE, DROP ON `lexicon`.* TO 'opus'@'localhost';
FLUSH PRIVILEGES;


* remove permissions:

REVOKE SELECT, INSERT, UPDATE, DELETE, CREATE, DROP ON `lexicon`.* FROM 'opus'@'localhost';
